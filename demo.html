// تهيئة المشغل const player = new PlayZ('#demo-player-1', { autoplay: false, muted: false, volume: 0.8, defaultQuality: '720p' }); // التحكم البرمجي document.getElementById('play-btn').addEventListener('click', () => { player.video.play(); }); document.getElementById('pause-btn').addEventListener('click', () => { player.video.pause(); }); document.getElementById('mute-btn').addEventListener('click', () => { player.video.muted = true; }); document.getElementById('unmute-btn').addEventListener('click', () => { player.video.muted = false; }); // تغيير الجودة document.getElementById('quality-576p').addEventListener('click', () => { player.setQuality('576p'); }); document.getElementById('quality-720p').addEventListener('click', () => { player.setQuality('720p'); }); document.getElementById('quality-1080p').addEventListener('click', () => { player.setQuality('1080p'); });
